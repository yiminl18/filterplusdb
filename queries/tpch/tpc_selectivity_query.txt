from
	part,
	supplier,
	partsupp,
	nation,
	region
where
	p_partkey = ps_partkey
	and s_suppkey = ps_suppkey
	and s_nationkey = n_nationkey
	and n_regionkey = r_regionkey
    and p_size < 10
    and ps_supplycost < 200
    and n_name = 'india'

select o_orderpriority,avg(l_extendedprice)
from
	orders, lineitem 
where
	o_orderdate >= '1992-01-01'
    and l_orderkey = o_orderkey
    and l_commitdate < l_receiptdate
	and o_orderdate < '1993-04-01'
    and l_shipdate < '1993-05-01' 
group by o_orderpriority
